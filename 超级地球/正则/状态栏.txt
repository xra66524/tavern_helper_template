  <script type="module">
    const get = (obj, path, def) => _.get(obj, path, def);

    // 辅助：根据完整度获取颜色 (0:Red -> 100:Green)
    function getHealthColor(val) {
      // Simple HSL interpolation: 0=0(Red), 100=120(Green)
      const hue = Math.max(0, Math.min(120, (val / 100) * 120));
      return `hsl(${hue}, 70%, 50%)`;
    }

    function populateCharacterData() {
      const all_variables = getAllVariables();
      const sd = get(all_variables, 'stat_data', {});

      // --- 1. 世界与光照 ---
      const world = get(sd, '世界', {});
      const time = get(world, '时空锚点', {});
      const weather = get(world, '环境场.气象', {});
      const loc = get(world, '环境场.地点信息', {});

      $('#time-str').text(`${time.年}年${time.月}月${time.日}日 ${time.周几} ${String(time.时).padStart(2,'0')}:${String(time.分).padStart(2,'0')}`);
      $('#weather-str').text(`${weather.天气} ${weather.温度}°C 💧${weather.湿度}%`);
      $('#loc-str').text(`${loc.国家}·${loc.城市}·${loc.区域}`);

      // 光照度动态背景
      const light = get(time, '光照度', 50);
      // 深蓝(HSL 220, 30%, 20%) -> 浅蓝(HSL 200, 80%, 60%)
      const h = 220 - (light / 100) * 20; // 220 -> 200
      const s = 30 + (light / 100) * 50;  // 30 -> 80
      const l = 20 + (light / 100) * 40;  // 20 -> 60
      $('.header-bar').css('background-color', `hsl(${h}, ${s}%, ${l}%)`);

      // --- 2. 用户基础 ---
      const user = get(sd, '用户', {});
      const base = get(user, '基础信息', {});
      $('#u-name').text(base.姓名);
      $('#u-nation').text(base.国籍);
      $('#u-job').text(base.职业);
      $('#u-class').text(base.社会阶层);
      $('#u-age').text(base.年龄);
      $('#u-desc').text(base.外貌特征);

      // 属性
      ['STR','CON','SIZ','DEX','APP','INT','POW','EDU','LUC'].forEach(k => {
        $(`#attr-${k}`).text(get(user, `属性.${k}`, 50));
      });

      // --- 3. 生理代谢 ---
      const meta = get(user, '生理代谢', {});

      // HP/MP/Libido
      const setBar = (id, cur, max, colorClass) => {
        const pct = Math.min(100, (cur/max)*100);
        $(`#${id}-text`).text(`${cur}/${max}`);
        $(`#${id}-bar`).css('width', `${pct}%`).attr('class', `progress-fill ${colorClass}`);
      };

      const hp = get(meta, '生命值', [10,10]);
      setBar('hp', hp[0], hp[1], 'bg-danger');

      const mp = get(meta, '精力值', [10,10]);
      setBar('mp', mp[0], mp[1], 'bg-success'); // Blue/Green

      const lib = get(meta, '性致', [0,50]);
      setBar('lib', lib[0], lib[1], 'bg-warning'); // Pink/Purple in CSS override? Using warning for now

      // 需求
      const needs = [
        {k:'饱食度',id:'hunger'},{k:'水分',id:'thirst'},{k:'洁净度',id:'hygiene'},
        {k:'压力值',id:'stress'},{k:'疲劳度',id:'fatigue'}
      ];
      needs.forEach(n => {
        const val = get(meta, n.k, 50);
        $(`#need-${n.id}-val`).text(val + '%');
        let color = 'bg-success';
        if (n.id==='stress'||n.id==='fatigue') {
          if(val>80) color='bg-danger'; else if(val>50) color='bg-warning';
        } else {
          if(val<20) color='bg-danger'; else if(val<50) color='bg-warning';
        }
        $(`#need-${n.id}-bar`).css('width', `${val}%`).attr('class', `progress-fill ${color}`);
      });

      // 排泄
      const toilet = get(meta, '排泄欲望', {});
      $('#need-poo').text(get(toilet, '便意', 0) + '%');
      $('#need-pee').text(get(toilet, '尿意', 0) + '%');

      // --- 4. 身体可视化 ---
      const body = get(user, '解剖学状态', {});
      const parts = {
        '头部': 'head', '颈部': 'neck', '躯干': 'torso',
        '左臂': 'larm', '右臂': 'rarm', '私处': 'groin',
        '左腿': 'lleg', '右腿': 'rleg'
      };

      let bodyTextHtml = '';
      Object.entries(parts).forEach(([zh, en]) => {
        const data = get(body, zh, {完整度:100, 状态标签:[], 感觉描述:'无'});
        const color = getHealthColor(data.完整度);
        $(`.bp-${en}`).css('background-color', color);

        bodyTextHtml += `
          <div class="body-text-item">
            <div class="bt-header">
              <span>${zh}</span>
              <span style="color:${color}">${data.完整度}%</span>
            </div>
            <div class="bt-desc">${data.状态标签.join(', ')} | ${data.感觉描述}</div>
          </div>
        `;
      });
      $('#body-text-list').html(bodyTextHtml);

      // --- 5. 认知与精神 ---
      const mind = get(user, '认知与精神', {});
      const mh = get(mind, '心理健康', {});
      $('#mind-status').text(mh.当前状态);
      $('#mind-res').text(mh.心理韧性);
      $('#mind-mood').text(get(mind, '心情', '待定'));
      $('#mind-thought').text(get(mind, '当前念头', '...'));
      $('#mind-trauma').text(get(mh, '心理创伤', []).join(', ') || '无');

      // --- 6. 数字身份 ---
      const digital = get(user, '数字身份', {});
      let digHtml = '';
      Object.entries(digital).forEach(([id, d]) => {
        digHtml += `
          <div class="digital-persona">
            <div class="dp-header"><span>${d.平台}</span><span>${d.网名}</span></div>
            <div class="dp-row"><span>粉丝: ${d.粉丝数}</span><span>风险: ${d.风险等级}</span></div>
            <div style="font-size:10px; color:#aaa">"${d.发布内容}"</div>
          </div>
        `;
      });
      if(!digHtml) digHtml = '<div style="font-size:12px;color:#999">无活跃数字身份</div>';
      $('#digital-list').html(digHtml);

      // --- 7. 着装 ---
      const attire = get(user, '当前着装', {});
      const attireMap = [
        {k:'头部',l:'头'}, {k:'眼部',l:'眼'}, {k:'颈部',l:'颈'},
        {k:'外套',l:'外'}, {k:'上身外层',l:'上外'}, {k:'上身内层',l:'上内'},
        {k:'手部',l:'手'}, {k:'下身外层',l:'下外'}, {k:'下身内层',l:'下内'},
        {k:'袜',l:'袜'}, {k:'鞋',l:'鞋'}, {k:'饰品',l:'饰'}
      ];
      let attireHtml = '';
      attireMap.forEach(i => {
        let val = get(attire, i.k, '无');
        if(Array.isArray(val)) val = val.join(', ');
        attireHtml += `
          <div class="equip-item">
            <span class="equip-slot">${i.k}</span>
            <div style="white-space:nowrap; overflow:hidden; text-overflow:ellipsis;">${val}</div>
          </div>
        `;
      });
      $('#attire-grid').html(attireHtml);

      // --- 8. 资产与负重 ---
      const inv = get(user, '物资与负重', {});
      const assets = get(inv, '资产', {});
      const curr = get(assets, '主货币', 'CNY');
      $('#money-cash').text(`${curr} ${assets.现金}`);
      $('#money-bank').text(`${curr} ${assets.电子存款}`);
      $('#money-debt').text(`${curr} ${assets.债务}`);
      $('#money-score').text(assets.信用评分);

      // 负重
      const load = get(inv, '负重', {});
      const loadPct = Math.min(100, (load.当前/load.上限)*100);
      let loadClass = 'bg-success';
      if(load.惩罚等级===1) loadClass = 'bg-yellow';
      if(load.惩罚等级===2) loadClass = 'bg-warning'; // Orange
      if(load.惩罚等级===3) loadClass = 'bg-danger';
      $('#load-text').text(`${load.当前} / ${load.上限} kg (Lv.${load.惩罚等级})`);
      $('#load-bar').css('width', `${loadPct}%`).attr('class', `progress-fill ${loadClass}`);

      // 装备栏
      const equip = get(inv, '背包系统.装备栏', {});
      $('#eq-main').text(equip.主手 || '空');
      $('#eq-off').text(equip.副手 || '空');
      $('#eq-pocket').text((equip.随身口袋||[]).join(', ') || '空');

      // 背包物品 (简单计算格数：每个物品占格x数量)
      const bagItems = get(inv, '背包系统.物品清单', {});
      let usedSlots = 0;
      let bagHtml = '';
      Object.entries(bagItems).forEach(([n, d]) => {
        const slots = (d.占格||1) * (d.数量||1);
        usedSlots += slots;
        bagHtml += `<li class="list-item" style="display:flex;justify-content:space-between"><span>${n} x${d.数量}</span><span style="color:#aaa">${d.重量}kg</span></li>`;
      });
      $('#bag-slots').text(`已用空间: ${usedSlots}格`);
      $('#bag-list').html(bagHtml || '<li class="list-item">背包空空如也</li>');

      // --- 9. 社会关系 ---
      const npcs = get(sd, '社交网络', {});
      let npcHtml = '';
      Object.entries(npcs).forEach(([id, d]) => {
        const profile = d.档案||{};
        const rel = d.关系||{};
        const cog = d.认知状态||{};
        npcHtml += `
          <li class="list-item">
            <div style="display:flex;justify-content:space-between;font-weight:600">
              <span>${profile.姓名||id} <span style="font-weight:400;color:#888">(${profile.职业})</span></span>
              <span style="color:var(--accent-color)">♥${rel.好感度}</span>
            </div>
            <div style="font-size:11px;color:#666;margin-top:2px;">
              关系:${rel.显性关系} | 信任:${rel.信任度} | 支配:${rel.支配值}
            </div>
            <div style="font-size:10px;color:#aaa;margin-top:1px;">
               ${cog.推测当前行为 || '未知行踪'}
            </div>
          </li>
        `;
      });
      $('#npc-list').html(npcHtml || '<li class="list-item">暂无社交记录</li>');

      // --- 10. 任务与信息流 ---
      const logs = get(sd, '事务日志', {});
      let questHtml = '';
      // 分离进行中和已结束
      const activeQ = [], doneQ = [];
      Object.entries(logs).forEach(([id, q]) => {
        if(q.状态==='进行中') activeQ.push(q); else doneQ.push(q);
      });

      if(activeQ.length) {
        questHtml += '<div style="font-weight:bold;margin:4px 0;">进行中</div>';
        activeQ.forEach(q => {
          questHtml += `<div class="quest-active"><div>${q.标题}</div><div style="font-size:11px;color:#666">${q.目标}</div></div>`;
        });
      }
      if(doneQ.length) {
        questHtml += '<div style="font-weight:bold;margin:4px 0;color:#999">历史记录</div>';
        doneQ.forEach(q => {
          questHtml += `<div class="quest-done"><div>${q.标题} (${q.状态})</div></div>`;
        });
      }
      $('#quest-list').html(questHtml || '<div class="list-item">无任务记录</div>');

      // 信息流
      const info = get(sd, '资讯流', []);
      let infoHtml = '';
      // 取最近10条，倒序
      info.slice().reverse().slice(0, 10).forEach(i => {
        let tagClass = 'info-tag-b';
        if(i.来源类型 && i.来源类型.includes('A类')) tagClass = 'info-tag-a';
        if(i.来源类型 && i.来源类型.includes('C类')) tagClass = 'info-tag-c';

        infoHtml += `
          <div class="info-item">
            <div class="info-meta">
              <span class="${tagClass}">[${i.来源类型.split(':')[0]}]</span>
              <span>${i.可信度}</span>
            </div>
            <div class="info-content">${i.标题}: ${i.内容}</div>
          </div>
        `;
      });
      $('#info-list').html(infoHtml || '<div class="list-item">暂无情报</div>');

      // 摘要
      const sum = get(sd, '摘要', []);
      $('#summary-text').text(sum[sum.length-1] || '暂无剧情摘要');
    }

    async function init() {
      await waitGlobalInitialized('Mvu');
      populateCharacterData();
      eventOn(Mvu.events.VARIABLE_UPDATE_ENDED, populateCharacterData);
    }
    $(errorCatched(init));
