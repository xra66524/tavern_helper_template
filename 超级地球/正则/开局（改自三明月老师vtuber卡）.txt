    <script>
      const STATS = [
        { id: 'STR', name: '力量', desc: '肉搏与负重' },
        { id: 'CON', name: '体质', desc: '健康与耐力' },
        { id: 'SIZ', name: '体型', desc: '身材与HP' },
        { id: 'DEX', name: '敏捷', desc: '速度与闪避' },
        { id: 'APP', name: '外貌', desc: '魅力与印象' },
        { id: 'INT', name: '智力', desc: '分析与逻辑' },
        { id: 'POW', name: '意志', desc: '精神抗性' },
        { id: 'EDU', name: '教育', desc: '知识水平' },
        { id: 'LUC', name: '幸运', desc: '机运' }
      ];

      // 初始化
      $(document).ready(() => {
        renderStats();
        randomizeAll(); // 默认随机一次
      });

      function renderStats() {
        const $container = $('#statsContainer');
        $container.empty();

        STATS.forEach(s => {
          const html = `
            <div class="stat-row">
              <div class="stat-label">${s.id}</div>
              <div class="stat-slider-container">
                <input type="range" class="stat-slider" id="in-${s.id}"
                       min="15" max="90" step="5"
                       oninput="$('#val-${s.id}').text(this.value)">
              </div>
              <div class="stat-value" id="val-${s.id}">50</div>
              <div class="stat-desc">${s.name}</div>
            </div>
          `;
          $container.append(html);
        });
      }

      function randomizeAll() {
        STATS.forEach(s => {
          // 3d6 * 5 模拟 (15-90)
          const roll = (Math.floor(Math.random() * 6) + 1 +
                        Math.floor(Math.random() * 6) + 1 +
                        Math.floor(Math.random() * 6) + 1) * 5;
          $(`#in-${s.id}`).val(roll);
          $(`#val-${s.id}`).text(roll);
        });

        // 清空文本框，保留选择框为随机
        $('#uName, #uAge, #uJob, #uBackground, #uLocation, #uMonth, #uDay').val('');
        $('#uGender, #uEconomy, #uSexExp').val('');
      }

      // 核心修复：增强版发送函数
      async function trySendToSillyTavern(text) {
        console.log("Attempting to send message:", text);
        let sent = false;

        // 1. 尝试 TavernHelper API (createChatMessages)
        if (!sent && window.parent && window.parent.createChatMessages) {
            try {
                await window.parent.createChatMessages([{ role: 'user', message: text }], {
                    insert_at: 'end',
                    refresh: 'affected'
                });
                console.log("Success: createChatMessages");
                sent = true;
            } catch (e) { console.error("Failed: createChatMessages", e); }
        }

        // 2. 尝试 executeSlashCommands (新版 API)
        if (!sent && window.parent && window.parent.executeSlashCommands) {
            try {
                // 使用 JSON.stringify 转义换行符，防止命令截断
                const safeText = JSON.stringify(text);
                await window.parent.executeSlashCommands(`/send ${safeText}`);
                console.log("Success: executeSlashCommands");
                sent = true;
            } catch (e) { console.error("Failed: executeSlashCommands", e); }
        }

        // 3. 尝试 triggerSlash (旧版 API)
        if (!sent && window.parent && window.parent.triggerSlash) {
            try {
                const safeText = JSON.stringify(text);
                await window.parent.triggerSlash(`/send ${safeText}`);
                console.log("Success: triggerSlash");
                sent = true;
            } catch (e) { console.error("Failed: triggerSlash", e); }
        }

        // 4. 尝试 DOM 注入 (最后的手段，模拟用户输入)
        if (!sent && window.parent && window.parent.document) {
            try {
                const doc = window.parent.document;
                // 查找输入框和发送按钮 (兼容不同UI版本)
                const input = doc.querySelector('#send_textarea') || doc.querySelector('textarea[id*="send"]');
                const sendBtn = doc.querySelector('#send_but') || doc.querySelector('div[id*="send_but"]');

                if (input && sendBtn) {
                    // 模拟完整的输入事件链，确保 Vue/React 状态更新
                    input.value = text;
                    input.dispatchEvent(new Event('input', { bubbles: true }));
                    input.dispatchEvent(new Event('change', { bubbles: true }));

                    // 延迟点击，等待状态同步
                    setTimeout(() => sendBtn.click(), 100);
                    console.log("Success: DOM Injection");
                    sent = true;
                }
            } catch (e) { console.error("Failed: DOM Injection", e); }
        }

        return sent;
      }

      async function submitGenesis() {
        const btn = $('.btn-primary');
        const originalText = btn.text();
        btn.text('正在写入现实...').prop('disabled', true);

        // 收集数据
        const data = {
          name: $('#uName').val() || '(请随机生成)',
          gender: $('#uGender').val() || '(请随机生成)',
          age: $('#uAge').val() || '(请随机生成)',
          job: $('#uJob').val() || '(根据属性生成)',
          economy: $('#uEconomy').val() || '(根据职业生成)',
          sexExp: $('#uSexExp').val() || '(根据设定生成)',
          location: $('#uLocation').val() || '(请随机生成)',
          month: $('#uMonth').val() || '(随机)',
          day: $('#uDay').val() || '(随机)',
          bg: $('#uBackground').val() || '(请结合属性与职业，创作一段合理的背景故事)'
        };

        const stats = {};
        STATS.forEach(s => stats[s.id] = parseInt($(`#in-${s.id}`).val()));

        // 构建核心指令
        const prompt = `[系统指令: 执行 MVU 初始化程序]
<user>已提交初始档案，请立即执行以下操作：

1. **设定补全**：
   - 姓名: ${data.name}
   - 性别: ${data.gender}
   - 年龄: ${data.age}
   - 职业: ${data.job}
   - 经济: ${data.economy}
   - 性经验: ${data.sexExp}
   - 初始地点: ${data.location}
   - 初始日期: ${data.month}月${data.day}日
   - 核心属性: ${JSON.stringify(stats)}

2. **背景撰写 (World Info)**：
   基于上述属性（如STR=${stats.STR}, INT=${stats.INT}）和用户偏好："${data.bg}"，
   撰写一段约300字的【<user>背景】，需解释其职业来源、性格成因及当前处境。
   *必须使用自然流畅的小说笔法，而非列表。*

3. **变量实体化 (Variable Update)**：
   输出 MVU 更新指令，将所有 '待定' (Pending) 状态替换为具体值。
   - 基础属性必须严格等于上述数值。
   - 衍生属性 (HP, 负重等) 必须由 Schema 自动计算。
   - 物品清单必须根据职业发放初始装备 (如: 医生=急救包, 侦探=手电筒)。
   - 当前着装必须根据职业和经济状况设定。
   - **世界.时间.日期** 必须更新为设定的日期（若随机则由你决定）。
   - **世界.环境.当前地点** 必须更新为设定的地点。

请直接以剧情开场或背景介绍开始，并在末尾附带变量更新代码块。`;

        // 执行发送
        const success = await trySendToSillyTavern(prompt);

        if (success) {
            btn.text('✅ 指令已发送');
            setTimeout(() => btn.text(originalText).prop('disabled', false), 2000);
        } else {
            console.error("All send methods failed.");
            navigator.clipboard.writeText(prompt);
            alert("⚠️ 自动发送失败（API未检测到）。\n\n指令已复制到剪贴板，请手动粘贴发送。");
            btn.text(originalText).prop('disabled', false);
        }
      }
    </script>
