  <script type="module">
    function populateArchiveData() {
      const all_vars = getAllVariables();
      const stat_data = _.get(all_vars, 'stat_data', {});

      /* --- 第一组：国际概况 --- */
      const time = _.get(stat_data, '世界.当前时间', '未定义');
      const date = _.get(stat_data, '世界.当前日期', '未定义');
      const global_event = _.get(stat_data, '国际情况.国际大事件', '暂无重大事件');

      $('#val-time-date').text(`${date} ${time}`);
      $('#val-global-event').text(global_event);

      const next_meeting = _.get(stat_data, '国际情况.下一个国际重大会议', {});
      const meeting_name = _.get(next_meeting, '名称', '无');
      const meeting_time = _.get(next_meeting, '举办时间', '未定');
      $('#val-next-meeting').text(`${meeting_name} (${meeting_time})`);

      const attitudes = _.get(stat_data, '国际情况.世界主要国家对日本态度', {});
      let attitudes_html = '';
      if (Object.keys(attitudes).length > 0) {
        Object.entries(attitudes).forEach(([country, data]) => {
          const support = _.get(data, '支持度', '未知');
          const demand = _.get(data, '政治诉求', '无');
          attitudes_html += `
            <div class="gov-list-item">
              <div class="gov-sub-title">${country}</div>
              <div style="font-size: 12px;">支持度: ${support} | 诉求: ${demand}</div>
            </div>`;
        });
      } else {
        attitudes_html = '<div style="color:#999; font-size:12px;">暂无外交数据</div>';
      }
      $('#list-global-attitudes').html(attitudes_html);

      /* --- 第二组：国内概况 --- */
      const csdc_rate = _.get(stat_data, '日本政府情况.日元对CSDC汇率', '未提供');
      const usd_rate = _.get(stat_data, '日本政府情况.日元对美元汇率', '未提供');
      const gdp = _.get(stat_data, '日本政府情况.国内GDP总值', '未提供');
      const unemployment = _.get(stat_data, '日本政府情况.失业率', '未提供');
      const public_support = _.get(stat_data, '日本政府情况.民意支持率', '未提供');

      $('#val-exchange-rate').text(`CSDC: ${csdc_rate} | USD: ${usd_rate}`);
      $('#val-gdp').text(gdp);
      $('#val-unemployment').text(`${unemployment}%`);
      $('#val-public-support').text(`${public_support}%`);

      const factions = _.get(stat_data, '日本国内情况.主要政治势力', {});
      let factions_html = '';
      if (Object.keys(factions).length > 0) {
        Object.entries(factions).forEach(([name, data]) => {
          const spectrum = _.get(data, '政治光谱', '未知');
          const demand = _.get(data, '政治诉求', '未知');
          const area = _.get(data, '主要势力范围', '未知');
          const support = _.get(data, '支持度', '未知');
          const leader = _.get(data, '领导人', '未知');

          let figures_html = '';
          const figures = _.get(data, '主要人物', {});
          Object.entries(figures).forEach(([fig_name, fig_desc]) => {
            figures_html += `<span class="gov-tag">${fig_name}(${fig_desc})</span>`;
          });

          factions_html += `
            <div class="gov-list-item">
              <div class="gov-sub-title">${name} (领导人: ${leader})</div>
              <div style="font-size: 12px; margin-bottom: 2px;">光谱: ${spectrum} | 支持度: ${support}</div>
              <div style="font-size: 12px; margin-bottom: 2px;">诉求: ${demand}</div>
              <div style="font-size: 12px; margin-bottom: 2px;">势力范围: ${area}</div>
              <div style="font-size: 12px; color: #555;">骨干: ${figures_html || '暂无'}</div>
            </div>`;
        });
      } else {
        factions_html = '<div style="color:#999; font-size:12px;">暂无政治势力数据</div>';
      }
      $('#list-domestic-factions').html(factions_html);

      /* --- 第三组：政府概况 --- */
      const cabinet_term = _.get(stat_data, '日本政府情况.本届内阁届数', '未知');
      const cabinet_duration = _.get(stat_data, '日本政府情况.本届内阁持续时间', '未知');
      const cabinet_task = _.get(stat_data, '日本政府情况.内阁目前任务', '无');

      $('#val-cabinet-term').text(`第 ${cabinet_term} 届`);
      $('#val-cabinet-duration').text(cabinet_duration);
      $('#val-cabinet-task').text(cabinet_task);

      const ministries = _.get(stat_data, '日本政府情况.各省厅', {});
      let ministries_html = '';
      if (Object.keys(ministries).length > 0) {
        Object.entries(ministries).forEach(([name, data]) => {
          const minister = _.get(data, '大臣', '空缺');
          const vice_minister = _.get(data, '副大臣', '空缺');
          const vice_sec = _.get(data, '事务次官', '空缺');
          const chief = _.get(data, '长官', '空缺');

          const task = _.get(data, '目前主要处理事务', '无');
          const imp = _.get(data, '重要程度', 0);
          const coop = _.get(data, '配合度', 0);
          const ctrl = _.get(data, '掌控度', 0);
          const staff = _.get(data, '现职员人数', 0);

          /* 逻辑判断：内阁府或xx省隐藏长官，xx厅隐藏事务次官 */
          let showChief = true;
          let showViceSec = true;

          if (name === '内阁府' || name.endsWith('省')) {
            showChief = false;
          }
          if (name.endsWith('厅')) {
            showViceSec = false;
          }

          /* 构建高层人员显示字符串 */
          let officials_html = `大臣: ${minister} | 副大臣: ${vice_minister}`;
          if (showViceSec) {
            officials_html += ` | 事务次官: ${vice_sec}`;
          }
          if (showChief) {
            officials_html += ` | 长官: ${chief}`;
          }

          ministries_html += `
            <div class="gov-list-item">
              <div class="gov-sub-title">${name}</div>
              <div style="font-size: 12px; margin-bottom: 2px; color: #8b0000;">${officials_html}</div>
              <div style="font-size: 12px; margin-bottom: 2px;">重要度: ${imp} | 配合度: ${coop} | 掌控度: ${ctrl} | 职员: ${staff}人</div>
              <div style="font-size: 12px; color: #555;">主要事务: ${task}</div>
            </div>`;
        });
      } else {
        ministries_html = '<div style="color:#999; font-size:12px;">暂无省厅数据</div>';
      }
      $('#list-ministries').html(ministries_html);

      /* --- 第四组：重要角色 --- */
      const characters = _.get(stat_data, '各角色情况', {});
      let chars_html = '';
      if (Object.keys(characters).length > 0) {
        Object.entries(characters).forEach(([name, data]) => {
          const position = _.get(data, '当前职位', '未知');
          const recent_affairs = _.get(data, '近期事务', '无');
          const task = _.get(data, '目前任务', '无');
          const health = _.get(data, '身体状况', '未知');

          let healthColor = '#333';
          if(health === '生病' || health === '亚健康') healthColor = '#d35400';
          if(health === '重病') healthColor = '#8b0000';

          chars_html += `
            <div class="gov-list-item">
              <div class="gov-sub-title">${name} <span style="font-size:11px; font-weight:normal; color:${healthColor};">[${health}]</span></div>
              <div style="font-size: 12px; margin-bottom: 2px;">职位: ${position}</div>
              <div style="font-size: 12px; margin-bottom: 2px;">近期事务: ${recent_affairs}</div>
              <div style="font-size: 12px; color: #555;">目前任务: ${task}</div>
            </div>`;
        });
      } else {
        chars_html = '<div style="color:#999; font-size:12px;">暂无角色数据</div>';
      }
      $('#list-characters').html(chars_html);
    }

    function toggleSection($header) {
      $header.parent('.gov-section').toggleClass('collapsed');
    }

    async function init() {
      await waitGlobalInitialized('Mvu');
      populateArchiveData();

      eventOn(Mvu.events.VARIABLE_UPDATE_ENDED, () => {
        populateArchiveData();
      });

      $('.gov-section-title').on('click', function () {
        toggleSection($(this));
      });
    }

    $(errorCatched(init));
  </script>
